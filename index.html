COMPONENTS

Reusability

var Hello = Vue.extend({
  template: '<div>{{ message }}</div>',
  props: ['myName'],
  data () {
    return {
      message: 'Hello ' + this.myName
    }
  }
})
// Register it with the tag <hello>
Vue.component('hello', Hello)
<hello my-name="James"></hello>

MODULARITY

Global-scope avoiding

// Hello.js
export default {
  template: '<div>{{ message }}</div>',
  props: ['myName'],
  data () {
    return {
      message: 'Hello ' + this.myName
    }
  }
}

MODULARITY

Global-scope avoiding

// main.js
import Hello from './Hello'
new Vue({
  el: '#example',
  data: object,
  components: { Hello }
})
<div id="example">
  <hello my-name="James"></hello>
</div>

Laravel integration

Gulp, elixir, bower etc

  • ./npm: Configurations, plugins, Elixir extensions
  • ./resources/bower: Bower libraries
  • ./resources/js: Js
  • ./resources/js/vue: VueJS
  • ./resources/js/vue/{group}: Front-end, Back-end
  • ./resources/js/vue/{group}/components: Vue components
  • ./resources/js/vue/{group}/services: Services
  • ./resources/js/vue/{group}/routes.js: Laravel routes on JS
  • ./resources/js/vue/{group}/your-main.js: Main file
